CREATE DATABASE NUTRIFLOW

USE NUTRIFLOW;
GO

CREATE TABLE CADASTRO (
    NOME NVARCHAR(50) NOT NULL,
    SOBRENOME NVARCHAR(50) NOT NULL,
    GENERO NVARCHAR(15) NOT NULL,
    EMAIL NVARCHAR(100) PRIMARY KEY,
    SENHA NVARCHAR(255) NOT NULL
);

CREATE TABLE METAS (
    EMAIL VARCHAR(100) PRIMARY KEY,
    HIDRATACAO VARCHAR(100),
    CONSUMO VARCHAR(100),
    EXERCICIO VARCHAR(100),
    PESO_ATUAL VARCHAR(100),
    PESO_DESEJADO VARCHAR(100),
    PERDA_PESO VARCHAR(5),
    GANHO_PESO VARCHAR(5),
    MELHORA_SONO VARCHAR(5),
    MELHORA_HIDRATACAO VARCHAR(5),
    MELHORA_ALIMENTACAO VARCHAR(5)
);
